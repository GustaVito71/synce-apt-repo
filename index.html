Configuración

# 1. Importar la clave GPG
wget -qO- https://gustavito71.github.io/synce-apt-repo/public.key | sudo tee /etc/apt/trusted.gpg.d/synce.asc

# 2. Agregar el repositorio (reemplazar CODENAME por bookworm, trixie, etc.)
echo "deb https://gustavito71.github.io/synce-apt-repo CODENAME main" | sudo tee /etc/apt/sources.list.d/synce.list

# 3. Actualizar e instalar
sudo apt update
sudo apt install synce-sync-engine synce-tools synce-dccm python3-rapi2
